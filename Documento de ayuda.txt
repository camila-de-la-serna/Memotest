var arr = [
  'https://juan-gallo.com/static/perfil2.jpg',
  'https://juan-gallo.com/static/perfil2.jpg'
]

var imgsLength = $('img').length

for (var i = 0; i < imgsLength; i++) {
  $('img').eq(i).attr('data-img', arr[i])
}

$('img').on('click', function() {
  var visible = $(this).attr('data-img')
  $(this).attr('src', visible)
})
